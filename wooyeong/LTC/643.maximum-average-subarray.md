function findMaxAverage(nums, k) {
    const n = nums.length;
    let sum = 0;
    for(let i = 0; i<k ;i++){
        sum+= nums[i];


    }
    let max = sum/k;
    let lpt = 0;
    let rpt = k  ;
    while(rpt < n){
        sum -= nums[lpt++];
        sum += nums[rpt++];
        console.log(sum)
        max = Math.max(max, sum/k)
    }
    return max;
};
