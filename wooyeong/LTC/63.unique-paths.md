```
var uniquePathsWithObstacles = function(obstacleGrid) {
    let count = 0;
    const m = obstacleGrid.length - 1;
    const n = obstacleGrid[0].length - 1;

  function dfs(rp, cp){
    if(rp > m || cp > n || obstacleGrid[rp][cp] === 1){
        return;
    }
    if(rp === m && cp ===n){
        count++;
        return
    }

    dfs(rp+1,cp);
    dfs(rp,cp+1);
  }
  dfs(0,0)
  return count
};

```
