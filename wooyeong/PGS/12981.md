function solution(n, words) {
    const len = words.length;
    const store = new Set();
    store.add(words[0]);
    let idx = 1;

    while(idx < len){
        const word = words[idx];

        const isExist =  store.has(word);
        const isDiff = words[idx-1].slice(-1) !== word.slice(0,1);

        if(isDiff || isExist){
            break;
        };

        store.add(word);
        idx++;

    }

    if(idx === len){
        return [0,0];
    }

    const order = (idx) % n +1;
    const people = Math.floor(idx / n) + 1;
    return [order,people];
}
